#include<iostream>
using namespace std;
char* StrCpy(char* str)
{
//1、计算src长度
	int len = 0;
	int i = 0;
	while (str[i] != '\0') {
		len++;
		i++;
	}
//2、用malloc申请长度为 len+1的内存空间
	char* str0= (char*)calloc((len + 1),sizeof(char));
	
//3、将src字符串逐个字符复制到新申请的内存空间
	for (int i = 0; i < len; i++)
		str0[i] = str[i];
//4、新申请内存空间的最后一位放入'\0'
	str0[len] = '\0';
//5、返回新申请的内存空间指针
	//free( str0);
	return str0;
}
int main() {
	char str[] = "abc";
	cout << StrCpy(str);
	return 0;
}
